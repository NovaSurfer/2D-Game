79 template <typename Component>
80 compId_t ECSComponentCreate(std::vector<uint8_t>& memory, EntityHandle entity,
81  BaseECSComponent* comp)
82 {
83  // Get last index
84  size_t index = memory.size();
85  // Resizing, making an offset with 'last index + component size'
86  memory.resize(index + Component::size);
87  // Construct new component in the address of 'memory' which is already allocated
88  Component* component = new(&memory[index]) Component(*(Component*)comp);
89  component->entity = entity;
90  return index;
91 }
